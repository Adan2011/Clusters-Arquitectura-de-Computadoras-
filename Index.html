// Ejemplo de análisis de complejidad
function buscarLineal(arr, valor) {
  for(let i = 0; i < arr.length; i++) {
    if(arr[i] === valor) return i;
  }
  return -1;
}
// Complejidad: O(n) - Lineal

class Nodo {
  constructor(dato) {
    this.dato = dato;
    this.siguiente = null;
  }
}

// Ejemplo de inserción al inicio
function insertarAlInicio(lista, dato) {
  let nuevoNodo = new Nodo(dato);
  nuevoNodo.siguiente = lista.cabeza;
  lista.cabeza = nuevoNodo;
}

// Ejemplo de ordenamiento por inserción
function ordenamientoInsercion(arr) {
  for(let i = 1; i < arr.length; i++) {
    let clave = arr[i];
    let j = i - 1;

    while(j >= 0 && arr[j] > clave) {
      arr[j + 1] = arr[j];
      j = j - 1;
    }
    arr[j + 1] = clave;
  }
  return arr;
}
// Complejidad: O(n²) en peor caso, O(n) en mejor caso
